// To heapify a subtree rooted with node i which is
// an index in arr[]. n is size of heap
void heapify(int arr[][2], int n, int i)
{
    int largest = i; // Initialize largest as root
    int l = 2*i + 1; // left = 2*i + 1
    int r = 2*i + 2; // right = 2*i + 2

    // If left child is larger than root
    if (l < n && arr[l][0] > arr[largest][0])
        largest = l;

    // If right child is larger than largest so far
    if (r < n && arr[r][0] > arr[largest][0])
        largest = r;

    // If largest is not root
    if (largest != i)
    {
        swap(arr[i], arr[largest]);

        // Recursively heapify the affected sub-tree
        heapify(arr, n, largest);
    }
}

void heapify(storm_event *arr[], int n, int i,bool state)
{
    int largest = i; // Initialize largest as root
    int l = 2*i + 1; // left = 2*i + 1
    int r = 2*i + 2; // right = 2*i + 2

    //check to see if fields are equal, if so sort by event id, otherwise sort by field
    if(state) {
        if(arr[l]->state == arr[largest]->state) {
            // If left child is larger than root
            if (l < n && arr[l]->event_id > arr[largest]->event_id)
                largest = l;

            // If right child is larger than largest so far
            if (r < n && arr[r]->event_id > arr[largest]->event_id)
                largest = r;
        } else {
            // If left child is larger than root
            if (l < n && arr[l]->state > arr[largest]->state)
                largest = l;

            // If right child is larger than largest so far
            if (r < n && arr[r]->state > arr[largest]->state)
                largest = r;
        }


    } else {
        if(arr[l]->month_name == arr[largest]->month_name) {
            // If left child is larger than root
            if (l < n && arr[l]->event_id > arr[largest]->event_id)
                largest = l;

            // If right child is larger than largest so far
            if (r < n && arr[r]->event_id > arr[largest]->event_id)
                largest = r;
        } else {
            // If left child is larger than root
            if (l < n && arr[l]->month_name > arr[largest]->month_name)
                largest = l;

            // If right child is larger than largest so far
            if (r < n && arr[r]->month_name > arr[largest]->month_name)
                largest = r;
        }

    }

    // If largest is not root
    if (largest != i)
    {
        swap(arr[i], arr[largest]);

        // Recursively heapify the affected sub-tree
        heapify(arr, n, largest, state);
    }
}

// main function to do heap sort
void heapSort(int arr[][2], int n)
{
    // Build heap (rearrange array)
    for (int i = n / 2 - 1; i >= 0; i--)
        heapify(arr, n, i);

    // One by one extract an element from heap
    for (int i=n-1; i>=0; i--)
    {
        // Move current root to end
        swap(arr[0], arr[i]);

        // call max heapify on the reduced heap
        heapify(arr, i, 0);
    }
}

void inOrderTraversal(bst *node, string high, string low, hash_table_entry *hashTable[], annual_storms *stormArray[], bool state) {
    //In order traversal is left subtree
    // root
    // Then Right subtree

    if(node != nullptr) {
        //Left Subtree recursion
        inOrderTraversal(node->left, high, low, hashTable, stormArray, state);

        //Check if the node is within bounds, if so then locate the storm event in the storm Array and print it's info
        if(high.compare(node->s) >= 0 && low.compare(node->s) <= 0) {
            hash_table_entry *ptr = hashTable[hashEventID(node->event_id)];

            while(ptr->event_id != node->event_id) {
                ptr = ptr->next;
            }

            int m;
            for (m = 0; ptr->year != stormArray[m]->year; m++) {}

            if(state) {
                cout << "State: " << stormArray[m]->events[ptr->event_index].state << " " << endl;
            } else {
                cout << "Month Name: " << stormArray[m]->events[ptr->event_index].month_name << " " << endl;
            }

            cout << "Event ID:   " << stormArray[m]->events[ptr->event_index].event_id << " "<< endl ;
            cout << "Year:       " << stormArray[m]->events[ptr->event_index].year << " " << endl;
            cout << "Event Type: " << stormArray[m]->events[ptr->event_index].event_type << endl;
            cout << "CZ Type:    " << stormArray[m]->events[ptr->event_index].cz_type << endl<< endl;

        }

        //Right Subtree recursion
        inOrderTraversal(node->right, high, low, hashTable,stormArray, state);
    }
}

bst* buildBST(storm_event *arr[], bst* root, int i, int n, bool state) {
    if(i < n) {
        bst* temp = new bst;
        temp->event_id = arr[i]->event_id;
        temp->left = nullptr;
        temp->right = nullptr;

        if(state) {
            temp->s = arr[i]->state;
        } else {
            temp->s = arr[i]->month_name;
        }

        root = temp;

        // insert left child
        root->left = buildBST(arr, root->left, 2 * i + 1, n,state);

        // insert right child
        root->right = buildBST(arr, root->right, 2 * i + 2, n,state);
    } return root;
}

int countBST(bst *root) {
    if(root) {
        return 1 + countBST(root->left) + countBST(root->right);
    } else {
        return 0;
    }
}

int findBSTHeight(bst *node) {
    {
        if (node == NULL)
            return 0;
        else
        {
            /* compute the depth of each subtree */
            int lDepth = findBSTHeight(node->left);
            int rDepth = findBSTHeight(node->right);

            /* use the larger one */
            if (lDepth > rDepth)
                return(lDepth + 1);
            else return(rDepth + 1);
        }
    }

}

void clearTree(bst* root)
{
    if (root == NULL) { return; }
    clearTree(root->left);
    clearTree(root->right);
    free(root);
}


int countMaxHeapHeight(int arr[][2], int n, int i) {

    if(i < n-1) {
        int l = 2*i + 1;
        int r = 2*i + 2;

        int lDepth = countMaxHeapHeight(arr, n,l);
        int rDepth = countMaxHeapHeight(arr, n,r);

        if (lDepth > rDepth)
            return(lDepth + 1);
        else return(rDepth + 1);
    } else {
        return 0;
    }


}

int countMaxHeap(int arr[][2], int n, int i) {
    if(i < n-1) {
        int l = 2*i + 1;
        int r = 2*i + 2;
        return 1 + countMaxHeap(arr,n,l) + countMaxHeap(arr,n,r);
    } else {
        return 0;
    }
}

                state = true;
            } else {
                state = false;
            }

            //Variable for deciding the size of the storm array to be cnostructed
            int bstcount;
            if (arguments[0] == "all") {
                bstcount = stormSize;
            } else {
                int m;
                for (m = 0; stoi(arguments[0]) != stormArray[m]->year; m++) {}

                bstcount = stormCount[m];
            }

            //Declare new array of storm events. This will be used to heapify and then to create the BST
            storm_event *stormData[bstcount];


            //Fill the stormdata array with the appriopriate storms
            int index = 0;
            if (arguments[0] == "all") {
                for (int j = 0; j < stormcount; j++) {
                    for (int l = 0; l < stormCount[j]; l++) {
                        stormData[index++] = &stormArray[j]->events[l];
                    }
                }
            } else {

                int m;
                for (m = 0; stoi(arguments[0]) != stormArray[m]->year; m++) {}

                for (int l = 0; l < stormCount[m]; l++) {
                    stormData[index++] = &stormArray[m]->events[l];
                }

            }

            //Heapify the stormData array to sort it
            heapify(stormData, bstcount, 0, state);

            //Build binary search tree from sorted stormData Array
            bst* root = buildBST(stormData, root, 0, bstcount, state);

            //Use recursive function to travel and print the values of the BST
            inOrderTraversal(root, arguments[3], arguments[2], hashTable, stormArray, state);

            cout << "Node Count: " << countBST(root) << endl;
            cout << "Root Height: " << findBSTHeight(root) << endl;
            cout << "Left Height: " << findBSTHeight(root->left) << endl;
            cout << "Right Height: " << findBSTHeight(root->right) << endl;
            clearTree(root);
        }
        cout << endl;

    }

    //Final Summary

    cout << endl << "========== Summary ==========" << endl;

    //Variable to store the count of chain lengths. Index corresponds to length. Max length is the storm size in the case all events get placed into the same chain
    int *chainlength = new int[stormSize];

    //Initialize chainlength array to 0
    for(int i =0;chainlength[i] = 0,i < stormSize;i++) {}

    for(int i = 0, length = 0; i< hashTableSize; i++,length=0) {
        hash_table_entry *ptr = hashTable[i];
        while(ptr != nullptr) {
            ptr = ptr->next;
            length++;
        }

        //Increase count of index at length by one to count the chain at that length
        chainlength[length]++;
    }

    cout << "Chain Length (N)        Number of chains with Length N" << endl;
    for(int i =0;i < stormSize,chainlength[i] > 0;i++) {
        cout << i << "                       " << chainlength[i] << endl;
    }

    float loadFactor = (float) stormSize/hashTableSize;
    cout << "Load factor: " << loadFactor << endl;
    cout << "";
    return 0;
