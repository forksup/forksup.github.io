struct graphList {
    int adjCoordinates;
    struct graphList * next;
    graphList(): adjCoordinates(0), next(nullptr) {}
};

struct A {
    int weight[2];
};


// Returns degree of ver in given graph
int findDegree(struct graphList * G) {
    int degree = 0;
    graphList * graphPtr = G;
    while (graphPtr != nullptr) {
        degree++;
        graphPtr = graphPtr->next;
    }

    return degree;
}

float clusteringCoefficient(int index, graphList * adj[3186]) {

    int count = 0;
    int degree = findDegree(adj[index]);
    vector < int > neighborIndex(degree);
    bool visited[degree][degree];

    for (int i = 0; i < degree; i++) {
        for (int l = 0; l < degree; l++) {
            visited[i][l] = false;
        }
    }

    graphList * ptr = adj[index];

    int l = 0;
    for (int l = 0; ptr != NULL; l++) {
        neighborIndex[l] = ptr->adjCoordinates;
        count++;
        ptr = ptr->next;
    }

    int neighborCount = 0;

    //node must at least degree of 2 to get clustering coeffecient
    if (!neighborIndex.empty() && degree > 1) {
        for (int i = 0; i < degree; i++) {
            graphList * ptr1 = adj[neighborIndex[i]];
            while (ptr1 != NULL) {
                for (int j = 0; j < degree; j++) {
                    if (neighborIndex[j] == ptr1 -> adjCoordinates && ptr1->adjCoordinates != index && visited[i][j] == false) {
                        neighborCount++;
                        visited[i][j] = true;
                        visited[j][i] = true;

                        break;
                    }
                }
                ptr1 = ptr1->next;
            }

        }

        return (float)(2 * neighborCount) / (degree * (degree - 1));
    } else {
        return 0;
    }

}

int DFSUtil(int index, bool visited[], graphList * adj[]) {
    // Mark the current node as visited and print it
    visited[index] = true;

    // Recur for all the vertices
    // adjacent to this vertex
    int count = 1;

    graphList * ptr = adj[index];

    while (ptr != nullptr) {
        if (!visited[ptr -> adjCoordinates]) {
            count += DFSUtil(ptr -> adjCoordinates, visited, adj);
        }
        ptr = ptr -> next;
    }

    return count;
}

// Connected Components function finds
std::vector < A > connectedComponents(graphList * adj[3186]) {
    vector < A > componentSize;

    bool * visited = new bool[3186]; // Fixed size known at compile time

    for (int v = 0; v < 3186; v++) {
        visited[v] = false;
    }

    for (int v = 0; v < 3186; v++) {
        if (!visited[v]) {

            int size = DFSUtil(v, visited, adj);

            bool found = false;

            for (int i = 0; i < componentSize.size(); i++) {
                if (componentSize[i].weight[0] == size) {
                    found = true;
                    componentSize[i].weight[1]++;
                    break;
                }
            }

            if (!found) {
                struct A sizeInfo = * new A;
                sizeInfo.weight[0] = size;
                sizeInfo.weight[1] = 1;

                if (componentSize.size() == 0) {
                    componentSize.push_back(sizeInfo);
                } else {
                    for (int i = 0; i < componentSize.size(); i++) {
                        if (size > componentSize[i].weight[0]) {
                            componentSize.insert(componentSize.begin() + i, sizeInfo);
                            break;
                        } else if (i == (componentSize.size() - 1)) {
                            componentSize.push_back(sizeInfo);
                            break;
                        }
                    }
                }
            }
        }

    }

    return componentSize;
}

//function for calculating Sxx and Syy
float calculateSD(float data[], float mean) {
    float standardDeviation = 0.0;

    for (int i = 0; i < 832; ++i)
        standardDeviation += pow(data[i] - mean, 2);

    return standardDeviation;
}
